\begin{exercise}{1.a} 
\begin{align*}
\int xe^{-x} dx = \int -x d(e^{-x}) &= -xe^{-x} + \int e^{-x} dx \\
\int xe^{-x} dx = -xe^{-x} - e^{-x} &= -e^{-x} (x+1)  \\
\int_{0}^{1} xe^{-x} dx = \frac{e-2}{e} &= \approx 0.26424
\end{align*}
\end{exercise}
